        function nextPage(pageNumber) {

        // Puzzle Logic

        let switches = [false, false, false];

        function toggleSwitch(num) {

            switches[num-1] = !switches[num-1];

            const btn = document.getElementById('sw' + num);

            btn.classList.toggle('on');

            

            checkPuzzle();

        }


        function checkPuzzle() {

            const status = document.getElementById('puzzle-status');

            const unlockBtn = document.getElementById('unlock-btn');

            

            // Logic: Switch 1 and 3 must be ON, Switch 2 must be OFF

            if (switches[0] === true && switches[1] === false && switches[2] === true) {

                status.innerText = "Logic: ACCESS GRANTED";

                status.style.color = "#00ff88";

                unlockBtn.style.display = "block";

            } else {

                status.innerText = "Logic: DENIED";

                status.style.color = "white";

                unlockBtn.style.display = "none";

            }

        }
